Build the new embedding port `src/lib/rag/` — structurally a mirror of `src/lib/llm/` (settings + provider adapters + one port entrypoint + a connectivity probe), but for embeddings only (OpenAI + Ollama; Anthropic has NO embeddings API, confirmed in the installed SDK).     Create `src/lib/rag/embed.ts` — the ONE embedding port entrypoint, mirroring complete()'s dispatch (relative imports only):
    ```typescript
    import { resolveEmbeddingProvider, embeddingModelId } from "./settings";
    import { embedOpenAi } from "./providers/openai-embed";
    import { embedOllama } from "./providers/ollama-embed";
    import { EMBEDDING_DIMENSIONS } from "./types";
    import type { SettingDb } from "./settings";

    export interface EmbedResult { embeddings: number[][]; provider: string; model: string; embeddingModel: string; }

    export async function embed(db: SettingDb, texts: string[]): Promise<EmbedResult> {
      const r = await resolveEmbeddingProvider(db);
      let embeddings: number[][];
      switch (r.provider) {
        case "openai": embeddings = await embedOpenAi({ apiKey: r.apiKey, model: r.model, input: texts }); break;
        case "ollama": embeddings = await embedOllama({ baseUrl: r.ollamaBaseUrl, model: r.model, input: texts }); break;
        default: throw new Error(`Unsupported embedding provider: ${r.provider}`);
      }
      // Loud failure on a wrong-dimension vector rather than a silent bad insert later (Anti-Pattern in research).
      for (const v of embeddings) {
        if (v.length !== EMBEDDING_DIMENSIONS) {
          throw new Error(`Embedding dimension mismatch: got ${v.length}, expected ${EMBEDDING_DIMENSIONS}`);
        }
      }
      return { embeddings, provider: r.provider, model: r.model, embeddingModel: embeddingModelId(r) };
    }
    ```
