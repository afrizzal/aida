    Create src/middleware.ts (EDGE-safe — NO Prisma import):
    ```ts
    import { type NextRequest, NextResponse } from "next/server";
    import { getSessionCookie } from "better-auth/cookies";

    const PUBLIC_PREFIXES = ["/login", "/setup", "/api/auth", "/api/health"];

    export function middleware(request: NextRequest) {
      const { pathname } = request.nextUrl;
      if (PUBLIC_PREFIXES.some((p) => pathname.startsWith(p))) return NextResponse.next();

      const sessionCookie = getSessionCookie(request);
      if (!sessionCookie) return NextResponse.redirect(new URL("/login", request.url));
      return NextResponse.next();
    }

    export const config = { matcher: ["/((?!_next/static|_next/image|favicon.ico).*)"] };
    ```
    Create tests/unit/middleware.test.ts:

    - Test C: middleware(new NextRequest("http://localhost/api/health")) -> assert passthrough.
