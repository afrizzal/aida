       export async function heartbeatHandler(_data?: unknown): Promise<void> {
         const now = new Date().toISOString();
         await prisma.systemSetting.upsert({
           where: { key: "heartbeat:lastRunAt" },
           update: { value: now },
           create: { key: "heartbeat:lastRunAt", value: now },
         });
       }

    2. Create `src/lib/worker/index.ts` — the standalone worker process (RELATIVE imports only):
       ```ts
       import PgBoss from "pg-boss";
       import { heartbeatHandler } from "./jobs/heartbeat";

       async function main() {
         const connectionString = process.env.DATABASE_URL;
         if (!connectionString) throw new Error("DATABASE_URL is required for the worker");
         const boss = new PgBoss(connectionString);
         boss.on("error", (err) => console.error("[worker] pg-boss error:", err));
         await boss.start();

         await boss.work("heartbeat", async ([job]) => {   // Pitfall 3: ARRAY destructuring is mandatory
           await heartbeatHandler(job.data);
         });
         await boss.schedule("heartbeat", "* * * * *", {}); // idempotent upsert; runs every minute

         console.log("[worker] started");

         const shutdown = async () => { await boss.stop(); process.exit(0); };
         process.on("SIGTERM", shutdown);
         process.on("SIGINT", shutdown);
       }

       main().catch((err) => { console.error("[worker] fatal:", err); process.exit(1); });
       ```
