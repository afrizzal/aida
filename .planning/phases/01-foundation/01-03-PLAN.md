    ```ts
    // Org-scoped DOMAIN models only. Better Auth models (User/Session/Organization/Member/...) and
    // global SystemSetting are intentionally EXCLUDED. Append new org-scoped models here in later phases.
    export const DOMAIN_MODELS = ["Setting"] as const;
    const isDomain = (model?: string) => !!model && (DOMAIN_MODELS as readonly string[]).includes(model);

    export function scopedDb(orgId: string) {
      if (!orgId) throw new Error("scopedDb requires a non-empty organizationId");
      return prisma.$extends({
        query: {
          $allModels: {
            async findMany({ model, args, query }) {
              if (isDomain(model)) args.where = { ...args.where, organizationId: orgId };
              return query(args);
            },
            async findFirst({ model, args, query }) {
              if (isDomain(model)) args.where = { ...args.where, organizationId: orgId };
              return query(args);
            },
            async count({ model, args, query }) {
              if (isDomain(model)) args.where = { ...args.where, organizationId: orgId };
              return query(args);
            },
            async create({ model, args, query }) {
              if (isDomain(model)) args.data = { ...args.data, organizationId: orgId };
              return query(args);
            },
            async update({ model, args, query }) {
              if (isDomain(model)) args.where = { ...args.where, organizationId: orgId };
              return query(args);
            },
            async updateMany({ model, args, query }) {
              if (isDomain(model)) args.where = { ...args.where, organizationId: orgId };
              return query(args);
            },
            async upsert({ model, args, query }) {
              if (isDomain(model)) {

              return query(args);

            async deleteMany({ model, args, query }) {
              if (isDomain(model)) args.where = { ...args.where, organizationId: orgId };
              return query(args);
            },
          },
        },
      });
    }
    ```

    ```ts
    import { headers } from "next/headers";
    import { redirect } from "next/navigation";
    import { auth } from "@/lib/auth";
    import { scopedDb } from "@/lib/scoped-db";

    export async function getCurrentSession() {
      return auth.api.getSession({ headers: await headers() });
    }

    export async function requireSession() {
      const session = await getCurrentSession();
      if (!session) redirect("/login");
      return session;
    }

    // Bridge: authenticated session -> active org -> tenant-scoped client.
    export async function getScopedDb() {
      const session = await requireSession();
      const orgId = session.session.activeOrganizationId;
      if (!orgId) throw new Error("No active organization on session — cannot scope queries");
      return { db: scopedDb(orgId), session, orgId };
    }
    ```
