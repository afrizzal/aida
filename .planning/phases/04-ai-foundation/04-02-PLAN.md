         /\bsk-(?:proj-)?[A-Za-z0-9_-]{20,}\b/g, // OpenAI-style API keys
         /\bsk-ant-[A-Za-z0-9_-]{20,}\b/g,       // Anthropic-style API keys

       import { resolveActiveProvider } from "./active-provider";
       import { redactSecrets } from "./redact";
       import { completeAnthropic } from "./providers/anthropic";
       import { completeOpenAi } from "./providers/openai";
       import { completeOllama } from "./providers/ollama";
       import type { CompleteParams, CompleteResult } from "./types";
       import type { SettingDb } from ... // reuse the Pick type

       export async function complete<T>(db: SettingDb, params: CompleteParams<T>): Promise<CompleteResult<T>> {
         const redactedPrompt = redactSecrets(params.prompt); // D-13/D-14 — no opt-out
         const s = await resolveActiveProvider(db);
         const base = { model: s.model, system: params.system, prompt: redactedPrompt, schema: params.schema };
         let output: T;
         switch (s.provider) {
           case "openai":    output = await completeOpenAi({ apiKey: s.apiKey, schemaName: params.schemaName, ...base }); break;
           case "anthropic": output = await completeAnthropic({ apiKey: s.apiKey, ...base }); break;
           case "ollama":    output = await completeOllama({ baseUrl: s.ollamaBaseUrl, ...base }); break;
           default: throw new Error(`Unsupported provider: ${s.provider}`);
         }
         return { output, redactedPrompt, provider: s.provider, model: s.model };
       }
